src/lib/src/utilities/types/aspect-ratio.types.ts

TsAspectRatioTypes

Define the allowed aspect ratios. Used in TsCardComponent
TsAspectRatioTypes: "16:9" | "4:3" | "3:2" | "5:4" | "1:1"

src/lib/src/utilities/types/button.types.ts

TsButtonActionTypes

Define the allowed TsButtonComponent action types
TsButtonActionTypes: "Button" | "Submit" | "Menu" | "Reset"
TsButtonFormatTypes

Define the allowed TsButtonComponent format types
TsButtonFormatTypes: "filled" | "hollow" | "collapsable"
TsButtonFunctionTypes

Define the allowed TsButtonComponent action types
TsButtonFunctionTypes: "button" | "search" | "submit"

src/lib/src/utilities/types/date.types.ts

TsDateTypes

Define the allowed date formats for the TsDatePipe.
TsDateTypes: "short" | "medium" | "extended" | "timestamp"

src/lib/src/utilities/types/input.types.ts

TsInputAutocompleteTypes

Define the allowed autocomplete variations for TsInputComponent

NOTE: This is not all valid types; only the ones this library supports.
TsInputAutocompleteTypes: "off" | "on" | "name" | "email" | "username" | "new-password" | "current-password" | "tel"
TsInputTypes

Define the allowed TsInputComponent input types
TsInputTypes: "text" | "password" | "email" | "hidden" | "number" | "search" | "tel" | "url"

src/lib/src/utilities/types/menu.types.ts

TsMenuPositionTypesX

Define the allowed X positions for a menu
TsMenuPositionTypesX: "before" | "after"
TsMenuPositionTypesY

Define the allowed Y positions for a menu
TsMenuPositionTypesY: "above" | "below"
TsMenuTriggerTypes

Define the allowed trigger types for a menu
TsMenuTriggerTypes: "default" | "utility"

src/lib/src/utilities/interfaces/navigation-item.interface.ts

TsNavigationItem

Define the allowed keys and types for an item passed to the TsNavigationComponent
TsNavigationItem: TsNavigationLinkItem | TsNavigationActionItem

src/lib/src/utilities/types/sort.types.ts

TsSortDirection

Define the allowed sort directions for {@link TsSort}
TsSortDirection: "asc" | "desc" | ""

src/lib/src/utilities/types/style-theme.types.ts

TsStyleThemeTypes

Define the allowed theme types
TsStyleThemeTypes: "primary" | "accent" | "warn"

src/lib/src/utilities/types/tooltip.types.ts

TsTooltipPositionTypes

Define the accepted string values for the TsTooltipComponent position
TsTooltipPositionTypes: "left" | "right" | "above" | "below" | "before" | "after"

src/lib/src/utilities/types/spacing.types.ts

TsVerticalSpacingTypes

Define the accepted string values for TsVerticalSpacingDirective
TsVerticalSpacingTypes: "small--2" | "small--1" | "small--0" | "none" | "default--0" | "large--0" | "large--1" | "large--2" | "large--3" | "large--4" | "large--5" | "large--6"
